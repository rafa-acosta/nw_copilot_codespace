! --- Management & Security (AAA) ---
hostname TEST-ROUTER-01
version 17.12
!
aaa new-model
aaa authentication login default local
aaa authorization exec default local 
aaa accounting exec default start-stop
!
username admin privilege 15 secret $9$L3x/I.1M$h8Xp6...
!
service password-encryption
no ip domain-lookup
ip domain-name test.local

! --- Network Infrastructure ---
interface Loopback0
 description Loopback-for-BGP-ID
 ip address 10.255.255.1 255.255.255.255
!
interface GigabitEthernet1
 description WAN-Primary-Link
 ip address 192.168.100.2 255.255.255.252
 no shutdown
!
interface GigabitEthernet2
 description LAN-Gateway
 ip address 10.1.1.1 255.255.255.0
 no shutdown

! --- Routing (BGP) ---
router bgp 65001
 bgp router-id 10.255.255.1
 bgp log-neighbor-changes
 neighbor 192.168.100.1 remote-as 65100
 !
 address-family ipv4
  network 10.1.1.0 mask 255.255.255.0
  neighbor 192.168.100.1 activate
 exit-address-family

! --- Access Control Lists (ACLs) ---
ip access-list extended MGMT-ACL
 permit tcp 10.0.0.0 0.255.255.255 host 10.255.255.1 eq 22
 deny ip any any log
!
ip access-list standard VTY-LIMIT
 permit 10.0.0.0 0.255.255.255

! --- System Management ---
line con 0
 exec-timeout 15 0
 stopbits 1
line vty 0 4
 access-class VTY-LIMIT in
 transport input ssh
!
end
